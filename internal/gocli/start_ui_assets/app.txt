  function clearUsageRefreshTimer() {
    if (!usageRefreshTimer) return;

  function shouldAutoRefreshUsage() {

    return state.usage.lastLoadedAt ? (Date.now() - state.usage.lastLoadedAt) : Number.POSITIVE_INFINITY;

  function syncUsageAutoRefresh() {
    if (!shouldAutoRefreshUsage()) {
      clearUsageRefreshTimer();
      return;
    }
    if (usageRefreshTimer) return;
    usageRefreshTimer = window.setInterval(() => {
      if (!shouldAutoRefreshUsage()) {
        clearUsageRefreshTimer();
        return;
      }
      if (state.usage.loading) return;
      if (!state.usage.loaded || usageRefreshAgeMs() >= USAGE_AUTO_REFRESH_INTERVAL_MS) {
        loadUsage({ silent: true });
      }
    }, USAGE_AUTO_REFRESH_INTERVAL_MS);
  }

  function refreshUsageIfStale() {
    if (!shouldAutoRefreshUsage() || state.usage.loading) {
      syncUsageAutoRefresh();
      return;
    }
    if (!state.usage.loaded || usageRefreshAgeMs() >= USAGE_AUTO_REFRESH_INTERVAL_MS) {
      loadUsage({ silent: true });
      return;
    }
    syncUsageAutoRefresh();
  }

  function api(path, options = {}) {
    const headers = {
      ...(options.body ? { "Content-Type": "application/json" } : {}),
      ...(options.headers || {}),
    };
    return fetch(apiBase + path, {
      ...options,
      headers,
    }).then(async (response) => {
      const contentType = response.headers.get("Content-Type") || "";
      const isJSON = contentType.includes("application/json");
      const payload = isJSON ? await response.json().catch(() => null) : null;
      const text = payload === null ? await response.text() : "";
      if (!response.ok) {
        const error = new Error((payload && payload.message) || text || response.statusText);
        if (payload && payload.code) error.code = payload.code;
        if (payload && payload.repair_command) error.repairCommand = payload.repair_command;
        throw error;
      }
      if (payload !== null) return payload;
      return response.json();
    });
  }

  function escapeHTML(value) {
    return String(value ?? "")
      .replace(/&/g, "&amp;")
      .replace(/</g, "&lt;")
      .replace(/>/g, "&gt;")
      .replace(/"/g, "&quot;")
      .replace(/'/g, "&#39;");
  }

  function defaultString(value, fallback) {
    const trimmed = String(value ?? "").trim();
    return trimmed || fallback;
  }

  function timestamp(value) {
    const parsed = Date.parse(value || "");
    return Number.isNaN(parsed) ? 0 : parsed;
  }

  function formatDateTime(value) {
    if (!value) return "No update yet";
    const date = new Date(value);
    if (Number.isNaN(date.getTime())) return value;
    return date.toLocaleString();
  }

  function formatTokenCompact(value) {
    const numeric = Number(value || 0);
    if (!Number.isFinite(numeric)) return "0";
    if (numeric >= 1000000) return `${(numeric / 1000000).toFixed(1)}M`;
    if (numeric >= 1000) return `${(numeric / 1000).toFixed(1)}k`;
    return String(Math.round(numeric));
  }

  function formatPercent(value) {
    const numeric = Number(value || 0);
    if (!Number.isFinite(numeric)) return "0%";
    return `${numeric.toFixed(1)}%`;
  }

  function toDateTimeLocal(value) {
    if (!value) return "";
    const date = new Date(value);
    if (Number.isNaN(date.getTime())) return "";
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, "0");
    const day = String(date.getDate()).padStart(2, "0");
    const hours = String(date.getHours()).padStart(2, "0");
    const minutes = String(date.getMinutes()).padStart(2, "0");
    return `${year}-${month}-${day}T${hours}:${minutes}`;
  }

  function fromDateTimeLocal(value) {
    if (!value) return "";
    const date = new Date(value);
    if (Number.isNaN(date.getTime())) return "";
    return date.toISOString();
  }

  function sameLocalDate(left, right) {
    return left.getFullYear() === right.getFullYear()
      && left.getMonth() === right.getMonth()
      && left.getDate() === right.getDate();
  }

  function schedulerTomorrowAnchor() {
    const date = new Date();
    date.setDate(date.getDate() + 1);
    date.setHours(9, 0, 0, 0);
    return date;
  }

  function schedulerNextMondayAnchor() {
    const date = new Date();
    const day = date.getDay();
    let delta = (8 - day) % 7;
    if (delta === 0) delta = 7;
    date.setDate(date.getDate() + delta);
    date.setHours(9, 0, 0, 0);
    return date;
  }

  function schedulerLaterAnchor() {
    const date = new Date();
    date.setDate(date.getDate() + 14);
    date.setHours(9, 0, 0, 0);
    return date;
  }

  function schedulerScheduleAtForBucket(bucketID) {
    switch (bucketID) {
      case "tomorrow":
        return schedulerTomorrowAnchor().toISOString();
      case "next-week":
        return schedulerNextMondayAnchor().toISOString();
      case "later":
        return schedulerLaterAnchor().toISOString();
      default:
        return "";
    }
  }

  function schedulerBucketForItem(item) {
    const value = defaultString(item && item.schedule_at, "");
    if (!value) return "ready-now";
    const scheduled = new Date(value);
    if (Number.isNaN(scheduled.getTime())) return "ready-now";
    const now = new Date();
    if (scheduled.getTime() <= now.getTime()) return "ready-now";
    const tomorrow = schedulerTomorrowAnchor();
    if (sameLocalDate(scheduled, tomorrow)) return "tomorrow";
    const nextWeek = schedulerNextMondayAnchor();
    if (scheduled.getTime() <= nextWeek.getTime()) return "next-week";
    return "later";
  }

  function schedulerBucketSortKey(bucketID) {
    const index = SCHEDULER_BUCKETS.findIndex((bucket) => bucket.id === bucketID);
    return index === -1 ? SCHEDULER_BUCKETS.length : index;
  }

  function schedulerIssueNumber(targetURL) {
    const match = String(targetURL || "").match(/\/issues\/(\d+)(?:\/)?$/);
    return match ? Number.parseInt(match[1], 10) : 0;

    if (value <= 1) return "high";
    if (value <= 3) return "medium";

  function ensureRepoScheduler(repo) {
    const repoSlug = repo ? repo.repo_slug : "";
    if (state.repoScheduler.repoSlug !== repoSlug) {
      state.repoScheduler = {
        ...defaultRepoSchedulerState(),
        repoSlug,
      };
    }
    reconcileRepoScheduler(repo);
  }

  function reconcileRepoScheduler(repo) {
    if (!repo) {
      state.repoScheduler = defaultRepoSchedulerState();
      return;
    }
    const scheduled = trackedIssuePlannedItems(repo);
    if (!state.repoScheduler.selectedPlannedItemID || !scheduled.some((item) => item.id === state.repoScheduler.selectedPlannedItemID)) {
      state.repoScheduler.selectedPlannedItemID = scheduled[0] ? scheduled[0].id : "";
    }
    state.repoScheduler.results = (state.repoScheduler.results || []).map((item) => {
      const planned = trackedIssuePlannedItemForURL(repo, item.target_url);
      return {
        ...item,
        scheduled: Boolean(planned),
        planned_item_id: planned ? planned.id : "",
        planned_state: planned ? planned.state : "",
        schedule_at: planned ? planned.schedule_at : "",
      };
    });
  }

  function trackedIssuePlannedItems(repo) {
    if (!repo || !repo.state) return [];
    return Object.values(repo.state.planned_items || {})
      .filter((item) => item.launch_kind === "tracked_issue" && (item.state === "queued" || item.state === "failed"))
      .sort((left, right) => {
        const leftBucket = schedulerBucketSortKey(schedulerBucketForItem(left));
        const rightBucket = schedulerBucketSortKey(schedulerBucketForItem(right));
        if (leftBucket !== rightBucket) return leftBucket - rightBucket;
        if ((left.priority ?? 5) !== (right.priority ?? 5)) return (left.priority ?? 5) - (right.priority ?? 5);
        return timestamp(right.updated_at) - timestamp(left.updated_at);
      });
  }

  function trackedIssuePlannedItemForURL(repo, targetURL) {
    if (!repo || !repo.state || !targetURL) return null;
    return trackedIssuePlannedItems(repo).find((item) => defaultString(item.target_url, "") === defaultString(targetURL, "")) || null;
  }

  function buildRouteHash() {
    const params = new URLSearchParams();
    params.set("view", state.currentView);
    if (state.currentView === "repo" && state.selectedRepo) {
      params.set("repo", state.selectedRepo);
      params.set("tab", state.repoTab);
    } else if (state.currentView === "feedback") {
      params.set("tab", state.feedbackTab);
    }
    if (state.currentView === "issues" && state.issueQueue.selectedID) {
      params.set("issue", state.issueQueue.selectedID);
    }
    if (state.currentView === "investigations" && state.investigations.selectedRunID) {
      params.set("investigation", state.investigations.selectedRunID);
    }
    if (state.currentView === "approvals" && state.approvals.selectedID) {
      params.set("approval", state.approvals.selectedID);
    }
    return params.toString();
  }

  function syncHashState() {
    if (!state.hashReady) return;
    const nextHash = buildRouteHash();
    const normalizedCurrent = String(window.location.hash || "").replace(/^#/, "");
    if (normalizedCurrent === nextHash) return;
    const targetURL = nextHash ? `#${nextHash}` : window.location.pathname;
    window.history.replaceState(null, "", targetURL);
  }

  function applyHashState() {
    const params = new URLSearchParams(String(window.location.hash || "").replace(/^#/, ""));
    const nextView = defaultString(params.get("view"), state.currentView);
    state.currentView = nextView;
    state.selectedRepo = defaultString(params.get("repo"), state.selectedRepo);
    state.repoTab = defaultString(params.get("tab"), nextView === "repo" ? state.repoTab : "overview");
    if (nextView === "feedback") {
      state.feedbackTab = defaultString(params.get("tab"), state.feedbackTab);
    }
    state.issueQueue.selectedID = defaultString(params.get("issue"), state.issueQueue.selectedID);
    state.investigations.selectedRunID = defaultString(params.get("investigation"), state.investigations.selectedRunID);
    state.approvals.selectedID = defaultString(params.get("approval"), state.approvals.selectedID);
  }

  function findQuickSwitchOption(value) {
    const input = String(value || "").trim().toLowerCase();
    if (!input) return null;
    return quickSwitchOptions().find((option) => option.value.toLowerCase() === input) || null;
  }

  function quickSwitchOptions() {
    const options = [
      { value: "Home", action: "view", target: "home" },
      { value: "Issues", action: "view", target: "issues" },
      { value: "Investigations", action: "view", target: "investigations" },
      { value: "Work", action: "view", target: "work" },
      { value: "Usage", action: "view", target: "usage" },
      { value: "Feedback", action: "view", target: "feedback" },
      { value: "Approvals", action: "view", target: "approvals" },
    ];
    sortedRepos().forEach((repo) => {
      options.push({ value: `Repo: ${repo.repo_slug}`, action: "repo", target: repo.repo_slug });
    });
    (state.overview ? state.overview.work_runs || [] : []).slice(0, 8).forEach((run) => {
      options.push({ value: `Run: ${run.run_id}`, action: "run", target: run.run_id });
    });
    (state.overview ? state.overview.work_items || [] : []).slice(0, 8).forEach((item) => {
      options.push({ value: `Work Item: ${item.id}`, action: "work-item", target: item.id });
    });
    return options;
  }

  function runQuickSwitch() {
    const input = document.getElementById("quick-switch-input");
    if (!input) return;
    const value = input.value;
    const option = findQuickSwitchOption(value);
    if (!option) {
      announce("Quick switch target not found.");
      return;
    }
    if (option.action === "view") {
      state.currentView = option.target;
      if (option.target !== "repo") {
        ensureViewData(option.target);
      }
      renderApp();
      input.value = "";
      return;
    }
    if (option.action === "repo") {
      state.currentView = "repo";
      state.selectedRepo = option.target;
      state.repoTab = "overview";
      renderApp();
      input.value = "";
      return;
    }
    if (option.action === "run") {
      openLogDrawer(option.target);
      input.value = "";
      return;
    }
    if (option.action === "work-item") {
      openWorkItemDrawer(option.target);
      input.value = "";
      return;
    }
  }

  function matchesAttentionFilter(item, value) {
    if (!value || value === "all") return true;
    return defaultString(item && item.attention_state, "") === value;
  }

  function matchesRepoFilter(item, value) {
    if (!value) return true;
    return defaultString(item && item.repo_slug, "") === value;
  }

  function matchesKindFilter(item, value) {
    if (!value || value === "all") return true;
    return defaultString(item && item.kind, defaultString(item && item.backend, "")) === value;
  }

  function matchesQuery(item, fields, query) {
    const normalized = String(query || "").trim().toLowerCase();
    if (!normalized) return true;
    return fields.some((field) => defaultString(item && item[field], "").toLowerCase().includes(normalized));
  }

  function buildFilterToolbar(view, options = {}) {
    const filter = filterState(view);
    const activeCount = activeFilterCount(filter);
    const repoOptions = ['<option value="">All repos</option>'].concat(sortedRepos().map((repo) => {
      const selected = filter.repo === repo.repo_slug ? "selected" : "";
      return `<option value="${escapeHTML(repo.repo_slug)}" ${selected}>${escapeHTML(repo.repo_slug)}</option>`;
    })).join("");
    const attentionOptions = [
      { value: "all", label: "All states" },
      { value: "failed", label: "Failed" },
      { value: "blocked", label: "Blocked" },
      { value: "active", label: "Active" },
      { value: "queued", label: "Queued" },
      { value: "completed", label: "Completed" },
    ].map((option) => {
      const selected = filter.attention === option.value ? "selected" : "";
      return `<option value="${escapeHTML(option.value)}" ${selected}>${escapeHTML(option.label)}</option>`;
    }).join("");
    const kindSelect = Array.isArray(options.kindOptions) && options.kindOptions.length
      ? `
        <label class="selector config-selector">
          <span>Kind</span>
          <select data-view-filter="${escapeHTML(view)}" data-view-filter-name="kind">
            ${options.kindOptions.map((option) => {
              const selected = filter.kind === option.value ? "selected" : "";
              return `<option value="${escapeHTML(option.value)}" ${selected}>${escapeHTML(option.label)}</option>`;
            }).join("")}
          </select>
        </label>
      `
      : "";
    return `

  const SCOUT_ROLE_ORDER = ["improvement", "enhancement", "ui"];
  const SCOUT_ROLE_META = {
    improvement: { label: "Improvement Scout", defaultSchedule: "when_resolved", defaultSessionLimit: 0 },
    enhancement: { label: "Enhancement Scout", defaultSchedule: "when_resolved", defaultSessionLimit: 0 },
    ui: { label: "UI Scout", defaultSchedule: "when_resolved", defaultSessionLimit: 4 },
  };

  function normalizeScoutRoleKey(roleKey) {
    return String(roleKey || "").replace("-scout", "");
  }

  function serverScoutCatalog(repo) {
    if (repo && Array.isArray(repo.scout_catalog) && repo.scout_catalog.length) return repo.scout_catalog;
    if (state.overview && Array.isArray(state.overview.scout_catalog) && state.overview.scout_catalog.length) return state.overview.scout_catalog;
    return [];
  }

  function repoScoutCatalog(repo) {
    const catalog = serverScoutCatalog(repo);
    if (catalog.length) return catalog;
    return SCOUT_ROLE_ORDER.map((roleKey) => ({
      role: `${roleKey}-scout`,
      config_key: roleKey,
      display_label: SCOUT_ROLE_META[roleKey].label,
      default_schedule: SCOUT_ROLE_META[roleKey].defaultSchedule,
      default_session_limit: SCOUT_ROLE_META[roleKey].defaultSessionLimit,
      supports_session_limit: SCOUT_ROLE_META[roleKey].defaultSessionLimit > 0,
    }));
  }

  function repoScoutMeta(roleKey, repo) {
    const key = normalizeScoutRoleKey(roleKey);
    const match = repoScoutCatalog(repo).find((entry) => entry.config_key === key);
    if (match) {
      return {
        label: defaultString(match.display_label, SCOUT_ROLE_META[key] ? SCOUT_ROLE_META[key].label : key),
        defaultSchedule: defaultString(match.default_schedule, SCOUT_ROLE_META[key] ? SCOUT_ROLE_META[key].defaultSchedule : "when_resolved"),
        defaultSessionLimit: Number(match.default_session_limit) > 0 ? Number(match.default_session_limit) : 0,
        supportsSessionLimit: Boolean(match.supports_session_limit),
      };
    }
    return SCOUT_ROLE_META[key] || SCOUT_ROLE_META.improvement;
  }

  function repoScoutConfigsByRole(repo) {
    if (repo && repo.scouts_by_role) return repo.scouts_by_role;
    const fallback = {};
    const scouts = repo && repo.scouts;
    repoScoutCatalog(repo).forEach((entry) => {
      fallback[entry.config_key] = scouts && scouts[entry.config_key] ? scouts[entry.config_key] : {};
    });
    return fallback;
  }

  function repoScoutSnapshot(role, config) {
    const labels = Array.isArray(config && config.labels) ? config.labels.filter((label) => String(label || "").trim()) : [];
    const meta = repoScoutMeta(role);
    const defaultSessionLimit = meta.defaultSessionLimit;
    return {
      enabled: Boolean(config && config.enabled),
      policy_path: defaultString(config && config.policy_path, ""),
      mode: defaultString(config && config.mode, "auto"),
      schedule: defaultString(config && config.schedule, defaultString(meta.defaultSchedule, "when_resolved")),
      issue_destination: defaultString(config && config.issue_destination, "local"),
      fork_repo: defaultString(config && config.fork_repo, ""),
      labels,
      session_limit: Number(config && config.session_limit) > 0 ? Number(config.session_limit) : defaultSessionLimit,
    };
  }

  function repoConfigSnapshot(repo) {
    if (!repo) return null;
    const issuePickMode = defaultString(repo.issue_pick_mode, "manual");
    const repoMode = defaultString(repo.repo_mode, "local");
    const scouts = {};
    repoScoutCatalog(repo).forEach((entry) => {
      scouts[entry.config_key] = repoScoutSnapshot(entry.config_key, repoScoutConfigsByRole(repo)[entry.config_key]);
    });
    const fixedScouts = {
      improvement: scouts.improvement || repoScoutSnapshot("improvement", {}),
      enhancement: scouts.enhancement || repoScoutSnapshot("enhancement", {}),
      ui: scouts.ui || repoScoutSnapshot("ui", {}),
    };
    return {
      repo_mode: repoMode,
      issue_pick_mode: issuePickMode,
      pr_forward_mode: defaultString(repo.pr_forward_mode, "approve"),
      fork_issues_mode: defaultString(repo.fork_issues_mode, issuePickModeToAutomationMode(issuePickMode)),
      implement_mode: defaultString(repo.implement_mode, issuePickModeToAutomationMode(issuePickMode)),
      publish_target: repoMode === "disabled"
        ? ""
        : defaultString(repo.publish_target, repoModeToPublishTarget(repoMode)),
      scout_catalog: repoScoutCatalog(repo),
      scouts_by_role: scouts,
      scouts: fixedScouts,
    };
  }

  function repoConfigSnapshotKey(config) {
    return JSON.stringify(config || {});
  }

  function cloneConfigSnapshot(config) {
    return config ? JSON.parse(JSON.stringify(config)) : null;
  }

  function prettyJSON(value) {
    return JSON.stringify(value || {}, null, 2);
  }

  function resetConfigEditorForRepo(repo) {
    const snapshot = repoConfigSnapshot(repo);
    state.configEditor = {
      ...defaultConfigEditor(),
      repoSlug: repo ? repo.repo_slug : "",
      server: snapshot,
      serverKey: repoConfigSnapshotKey(snapshot),
      draft: cloneConfigSnapshot(snapshot),
    };
  }

  function ensureConfigEditor(repo) {
    if (!repo) {
      state.configEditor = defaultConfigEditor();
      return;
    }
    const snapshot = repoConfigSnapshot(repo);
    const serverKey = repoConfigSnapshotKey(snapshot);
    if (state.configEditor.repoSlug !== repo.repo_slug) {
      resetConfigEditorForRepo(repo);
      return;
    }
    state.configEditor.server = snapshot;
    if (!state.configEditor.dirty) {
      state.configEditor.serverKey = serverKey;
      state.configEditor.draft = cloneConfigSnapshot(snapshot);
      state.configEditor.remoteChanged = false;
      return;
    }
    if (serverKey !== state.configEditor.serverKey) {
      state.configEditor.remoteChanged = true;
    }
  }

  function ensureScoutItemsState(repo) {

      state.scoutItems = defaultScoutItemsState();
      return;

  function selectedScoutItem() {
    if (!state.scoutItems.selectedItemID) return state.scoutItems.items[0] || null;
    return state.scoutItems.items.find((item) => item.id === state.scoutItems.selectedItemID) || state.scoutItems.items[0] || null;
  }

  function selectedScoutItemIDs() {
    return (state.scoutItems.selectedItemIDs || []).filter((itemID) => state.scoutItems.items.some((item) => item.id === itemID));
  }

  function allScoutItemIDs() {
    return state.scoutItems.items
      .map((item) => String(item.id || ""))
      .filter(Boolean);
  }

  function setSelectedScoutItemIDs(itemIDs) {
    const allowed = new Set(state.scoutItems.items.map((item) => item.id));
    const seen = new Set();
    state.scoutItems.selectedItemIDs = (itemIDs || []).filter((itemID) => {
      const id = String(itemID || "");
      if (!allowed.has(id) || seen.has(id)) return false;
      seen.add(id);
      return true;
    });
  }

  function selectAllScoutItems() {
    const itemIDs = allScoutItemIDs();
    setSelectedScoutItemIDs(itemIDs);
    if (itemIDs.length && !itemIDs.includes(state.scoutItems.selectedItemID)) {
      state.scoutItems.selectedItemID = itemIDs[0];
    }
  }

  function clearScoutItemSelection() {
    setSelectedScoutItemIDs([]);
  }

  function toggleScoutItemSelection(itemID, checked) {
    const current = new Set(selectedScoutItemIDs());
    if (checked) {
      current.add(itemID);
    } else {
      current.delete(itemID);
    }
    setSelectedScoutItemIDs(Array.from(current));
  }

  function scoutActionableSelection(action) {
    return selectedScoutItemIDs().filter((itemID) => {
      const item = state.scoutItems.items.find((entry) => entry.id === itemID);
      return item && (item.available_actions || []).includes(action);
    });
  }

  function applyScoutItemsPayload(repoSlug, payload) {
    if (payload.repo) {
      mergeRepoSummary(payload.repo, { rerender: false, resetScouts: false });
    }
    state.scoutItems.repoSlug = repoSlug;
    state.scoutItems.items = payload.items || [];
    state.scoutItems.loading = false;
    state.scoutItems.loaded = true;
    state.scoutItems.error = "";
    state.scoutItems.lastRepoUpdatedAt = payload.repo ? defaultString(payload.repo.updated_at, "") : state.scoutItems.lastRepoUpdatedAt;
    if (!state.scoutItems.selectedItemID || !state.scoutItems.items.some((item) => item.id === state.scoutItems.selectedItemID)) {
      state.scoutItems.selectedItemID = state.scoutItems.items[0] ? state.scoutItems.items[0].id : "";
    }
    setSelectedScoutItemIDs(state.scoutItems.selectedItemIDs || []);
  }

  function loadScoutItems(repoSlug, options = {}) {
    if (!repoSlug) return Promise.resolve();
    state.scoutItems.repoSlug = repoSlug;
    state.scoutItems.loading = true;
    if (!options.silent) {
      state.scoutItems.error = "";
    }
    if (!options.silent || !state.scoutItems.items.length) {
      renderRepoScoutsPage(selectedRepoSummary());
    }
    return api(`/api/v1/repos/${repoSlug}/scout-items`).then((payload) => {
      if (state.scoutItems.repoSlug !== repoSlug) return;
      applyScoutItemsPayload(repoSlug, payload);
      renderRepoScoutsPage(selectedRepoSummary());
    }).catch((error) => {
      if (state.scoutItems.repoSlug !== repoSlug) return;
      state.scoutItems.loading = false;
      state.scoutItems.error = error.message;
      renderRepoScoutsPage(selectedRepoSummary());
      if (!options.silent) {
        toast("danger", "Scout items failed", error.message);
      }
    });
  }

  function mutateScoutItem(repoSlug, itemID, action) {
    return api(`/api/v1/repos/${repoSlug}/scout-items/${encodeURIComponent(itemID)}/${action}`, {
      method: "POST",
    }).then((payload) => {
      applyScoutItemsPayload(repoSlug, payload);
      renderRepoScoutsPage(selectedRepoSummary());
      return payload;
    }).catch((error) => {
      toast("danger", "Scout item action failed", error.message);
      throw error;
    });
  }

  function mutateScoutItemsBulk(repoSlug, itemIDs, action) {
    const selectedBefore = new Set(selectedScoutItemIDs());
    return api(`/api/v1/repos/${repoSlug}/scout-items/batch`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        action,
        item_ids: itemIDs,
      }),
    }).then((payload) => {
      applyScoutItemsPayload(repoSlug, payload);
      const successfulIDs = new Set((payload.results || [])
        .filter((result) => result.status === "ok")
        .map((result) => result.item_id));
      const nextSelection = Array.from(selectedBefore).filter((itemID) => !successfulIDs.has(itemID));
      setSelectedScoutItemIDs(nextSelection);
      renderRepoScoutsPage(selectedRepoSummary());
      return payload;
    }).catch((error) => {
      toast("danger", "Scout bulk action failed", error.message);
      throw error;
    });
  }

  function usageQueryString() {
    const params = new URLSearchParams();
    const filters = state.usage.filters || {};
    if (filters.since && filters.since !== "all") params.set("since", filters.since);
    if (filters.root && filters.root !== "all") params.set("root", filters.root);
    if (filters.activity) params.set("activity", filters.activity);
    if (filters.phase) params.set("phase", filters.phase);
    if (filters.model) params.set("model", filters.model);
    return params.toString();
  }

  function ensureViewData(view) {
    if ((!state.overview || !Array.isArray(state.overview.repos) || !state.overview.repos.length) && !state.repoList.loaded && !state.repoList.loading) {
      loadRepos({ silent: true });
    }
    switch (view) {
      case "issues":
        if (!state.issueQueue.loaded && !state.issueQueue.loading) {
          loadIssueQueue({ silent: true });
        }
        break;
      case "investigations":
        if (!state.investigations.loaded && !state.investigations.loading) {
          loadInvestigations({ silent: true });
        }
        break;
      case "feedback":
        if (!state.feedback.loaded && !state.feedback.loading) {
          loadFeedback({ silent: true });
        }
        break;
      case "approvals":
        if (!state.approvals.loaded && !state.approvals.loading) {
          loadApprovals({ silent: true });
        }
        break;
      case "usage":
        {
          const liveUsageVersion = defaultString(state.liveVersions.usage, "");
          const currentUsageVersion = defaultString(state.usage.report && state.usage.report.version, "");
          const staleUsage = liveUsageVersion && liveUsageVersion !== currentUsageVersion;
          if ((!state.usage.loaded || staleUsage) && !state.usage.loading) {
            loadUsage({ silent: true });
          }
        }
        break;
      default:
        break;
    }
  }

      case "investigations":
        return loadInvestigations(options);
      case "feedback":
        return loadFeedback(options);
      case "approvals":
        return loadApprovals(options);
      case "usage":
        return loadUsage(options);
      default:
        return Promise.resolve();

  function currentRunSummary(runID) {
    if (!state.overview) return null;
    return (state.overview.work_runs || []).find((run) => run.run_id === runID) || null;
  }

  function currentWorkItemSummary(itemID) {

    return (state.overview.work_items || []).find((item) => item.id === itemID) || null;

  function sortRuns(runs) {
    return [ ...runs ].sort((left, right) => {
      const leftRank = attentionOrder[left.attention_state] ?? attentionOrder.active;
      const rightRank = attentionOrder[right.attention_state] ?? attentionOrder.active;
      if (leftRank !== rightRank) return leftRank - rightRank;
      const updatedDiff = timestamp(right.updated_at) - timestamp(left.updated_at);
      if (updatedDiff !== 0) return updatedDiff;
      return left.run_id.localeCompare(right.run_id);
    });
  }

  function pendingRuns() {
    if (!state.overview) return [];
    return sortRuns((state.overview.work_runs || []).filter((run) => run.pending));
  }

  function sortWorkItems(items) {
    return [ ...(items || []) ].sort((left, right) => {
      const leftRank = attentionOrder[left.attention_state] ?? attentionOrder.active;
      const rightRank = attentionOrder[right.attention_state] ?? attentionOrder.active;
      if (leftRank !== rightRank) return leftRank - rightRank;
      const updatedDiff = timestamp(right.updated_at) - timestamp(left.updated_at);
      if (updatedDiff !== 0) return updatedDiff;
      return left.id.localeCompare(right.id);
    });
  }

  function pendingWorkItems() {
    if (!state.overview) return [];
    return sortWorkItems((state.overview.work_items || []).filter((item) => item.pending));
  }

  function pendingRunsForRepo(repoSlug) {
    return pendingRuns().filter((run) => run.repo_slug === repoSlug);
  }

  function pendingWorkItemsForRepo(repoSlug) {
    return pendingWorkItems().filter((item) => item.repo_slug === repoSlug);
  }

  function repoScoutQueueCount(repo) {
    return (repo.scout_job_counts.queued || 0) +
      (repo.scout_job_counts.running || 0) +
      (repo.scout_job_counts.failed || 0);
  }

  function repoDismissedScoutCount(repo) {
    return repo.scout_job_counts.dismissed || 0;
  }

  function repoManualPlannedQueueCount(repo) {
    return (repo.planned_item_counts.queued || 0) +
      (repo.planned_item_counts.launching || 0);
  }

  function outstandingRepoCount(repo) {
    return (repo.issue_counts.queued || 0) +
      (repo.issue_counts.blocked || 0) +
      (repo.service_task_counts.queued || 0) +
      (repo.service_task_counts.running || 0) +
      repoScoutQueueCount(repo) +
      repoManualPlannedQueueCount(repo) +
      pendingRunsForRepo(repo.repo_slug).length +
      pendingWorkItemsForRepo(repo.repo_slug).length;
  }

  function closeConfirmationDialog() {
    const dialog = document.getElementById("confirm-dialog");
    if (dialog && dialog.open) {
      dialog.close("cancel");
    }
    const selector = state.confirmDialog.returnFocusSelector;
    const focusID = state.confirmDialog.returnFocusID;
    state.confirmDialog = defaultConfirmDialog();
    if (selector) {

  function renderConfirmDialog() {
    const dialog = document.getElementById("confirm-dialog");
    if (!dialog) return;
    const title = document.getElementById("confirm-dialog-title");
    const message = document.getElementById("confirm-dialog-message");
    const confirm = document.getElementById("confirm-dialog-confirm");
    if (!title || !message || !confirm) return;
    title.textContent = defaultString(state.confirmDialog.title, "Confirm Action");
    message.textContent = defaultString(state.confirmDialog.message, "Review this action before continuing.");
    confirm.textContent = defaultString(state.confirmDialog.confirmLabel, "Confirm");
    confirm.className = state.confirmDialog.tone === "danger" ? "tx-btn tx-btn-danger" : "tx-btn tx-btn-primary";
    if (state.confirmDialog.open && !dialog.open) {
      dialog.showModal();
      confirm.focus();
      return;
    }
    if (!state.confirmDialog.open && dialog.open) {
      dialog.close("cancel");
    }
  }

  function renderStatePill(attentionState, status) {

  }

  function pauseStateSummary(pauseUntil, pauseReason) {
    const until = String(pauseUntil || "").trim();
    const reason = String(pauseReason || "").trim();
    const parts = [];
    if (until) {
      const formatted = formatDateTime(until);
      parts.push(formatted || until);
    }
    if (reason) {
      parts.push(reason);
    }
    return parts.join(" · ");
  }

  function renderStatePillWithPause(attentionState, status, pauseUntil, pauseReason) {
    const detail = pauseStateSummary(pauseUntil, pauseReason);
    if (!detail) {
      return renderStatePill(attentionState, status);
    }
    return `${renderStatePill(attentionState, status)}<span class="data-value-text">${escapeHTML(detail)}</span>`;
  }

  function dropRepo(repoSlug) {
    const slug = String(repoSlug || "").trim();
    if (!slug) return;
    requestConfirmation({
      title: `Drop ${slug}?`,
      message: "This removes tracked repo state, queued work, and managed sandboxes from the workspace.",
      confirmLabel: "Drop Repo",
      tone: "danger",
      onConfirm() {
        api(`/api/v1/repos/${slug}/drop`, {
          method: "POST",
        }).then(() => {
          if (state.logDrawer) {
            state.logDrawer.close();
          }
          if (state.workItemDrawer) {
            state.workItemDrawer.close();
          }
          const refreshes = [load()];
          if (state.issueQueue.loaded) {
            refreshes.push(loadIssueQueue({ silent: true }));
          }
          if (state.investigations.loaded) {
            refreshes.push(loadInvestigations({ silent: true }));
          }
          if (state.feedback.loaded) {
            refreshes.push(loadFeedback({ silent: true }));
          }
          if (state.approvals.loaded) {
            refreshes.push(loadApprovals({ silent: true }));
          }
          Promise.allSettled(refreshes);
          toast("success", "Repo dropped", slug);
        }).catch((error) => {
          toast("danger", "Repo drop failed", error.message);
        });
      },
    });
  }

  function interpolateTemplate(template, row) {
    return String(template || "").replace(/\{\{\s*([^}\s]+)\s*\}\}/g, (_, key) => {
      const value = row && Object.prototype.hasOwnProperty.call(row, key) ? row[key] : "";
      return escapeHTML(value);
    });
  }

  function rowSelectionMeta(row, options) {
    if (typeof options.rowSelect !== "function") {
      return null;
    }
    return options.rowSelect(row);
  }

  function renderGridCell(column, row, view) {
    if (typeof column.render === "function") {
      return column.render(row, view);
    }
    if (column.template) {
      return interpolateTemplate(column.template, row);
    }
    return `<span class="data-value-text">${escapeHTML(defaultString(row[column.field], "-"))}</span>`;
  }

  function renderGrid(target, columns, data, options = {}) {
    const host = document.getElementById(target);
    if (!host) return;
    host.innerHTML = "";
    if (!columns.length || !data.length) {
      host.innerHTML = buildInlineNotice(options.emptyMessage || "No data yet");
      return;
    }
    const tableRows = data.map((row) => {
      const selection = rowSelectionMeta(row, options);
      const attrs = selection
        ? ` class="data-row-selectable ${selection.selected ? "data-row-selected" : ""}" tabindex="0" role="button" data-row-select-kind="${escapeHTML(selection.kind)}" data-row-select-id="${escapeHTML(selection.value)}" aria-pressed="${selection.selected ? "true" : "false"}"`
        : "";
      return `
        <tr${attrs}>
          ${columns.map((column) => `<td>${renderGridCell(column, row, "table")}</td>`).join("")}
        </tr>
      `;
    }).join("");
    const cardRows = data.map((row) => {
      const selection = rowSelectionMeta(row, options);
      const detailRows = columns.filter((column) => !column.action).map((column) => {
        return `
          <div class="data-card-row">
            <span class="data-card-label">${escapeHTML(column.label)}</span>
            <div class="data-card-value">${renderGridCell(column, row, "card")}</div>
          </div>
        `;
      }).join("");
      const actionRows = columns.filter((column) => column.action).map((column) => renderGridCell(column, row, "card")).join("");
      return `
        <article class="data-card ${selection ? `data-card-selectable ${selection.selected ? "data-card-selected" : ""}` : ""}" ${selection ? `tabindex="0" role="button" data-row-select-kind="${escapeHTML(selection.kind)}" data-row-select-id="${escapeHTML(selection.value)}" aria-pressed="${selection.selected ? "true" : "false"}"` : ""}>
          ${detailRows}
          ${actionRows ? `<div class="data-card-actions">${actionRows}</div>` : ""}
        </article>
      `;
    }).join("");
    host.innerHTML = `
      <div class="data-panel">
        <div class="data-table-wrap" style="max-height: ${escapeHTML(options.maxHeight || "420px")};">
          <table class="data-table">
            <thead>
              <tr>${columns.map((column) => `<th>${escapeHTML(column.label)}</th>`).join("")}</tr>
            </thead>
            <tbody>${tableRows}</tbody>
          </table>
        </div>
        <div class="data-cards">${cardRows}</div>
      </div>
    `;
  }

  function renderPendingJobsChart(target, runs) {
    const host = document.getElementById(target);
    if (!host) return;
    const grouped = {};
    runs.forEach((run) => {
      const key = runRepoDisplay(run);
      grouped[key] = (grouped[key] || 0) + 1;
    });
    const data = Object.entries(grouped)
      .map(([label, count]) => ({ label, count }))
      .sort((left, right) => right.count - left.count || left.label.localeCompare(right.label));
    if (!data.length) {
      destroyWidget(state.widgets[target]);
      delete state.widgets[target];
      delete state.widgetStateKeys[target];
      host.innerHTML = '<div class="status-pill">No pending jobs are active.</div>';
      return;
    }
    const chartOptions = {
      type: "bar",
      height: 300,
      title: "Pending jobs by repo",
      xAxis: { title: "Repo" },
      yAxis: { title: "Jobs" },
      series: [
        {
          name: "Pending jobs",
          data: data.map((item) => ({ x: item.label, y: item.count })),
        },
      ],
    };
    const nextKey = JSON.stringify(chartOptions.series);
    if (state.widgets[target] && state.widgetStateKeys[target] === nextKey) {
      return;
    }
    if (state.widgets[target] && typeof state.widgets[target].update === "function") {
      state.widgets[target].update(chartOptions);
      state.widgetStateKeys[target] = nextKey;
      return;
    }
    replaceWidget(target, Teryx.chart(`#${target}`, chartOptions));
    state.widgetStateKeys[target] = nextKey;
  }

  function renderUsageChart(target, chartOptions, emptyMessage) {

      delete state.widgetStateKeys[target];

      return;

    if (state.widgets[target] && state.widgetStateKeys[target] === nextKey) {
      return;
    }
    if (state.widgets[target] && typeof state.widgets[target].update === "function") {
      state.widgets[target].update(chartOptions);
      state.widgetStateKeys[target] = nextKey;
      return;
    }
    replaceWidget(target, Teryx.chart(`#${target}`, chartOptions));
    state.widgetStateKeys[target] = nextKey;
  }

  function filteredIssueQueueItems() {
    const filter = filterState("issues");
    return (state.issueQueue.items || []).filter((item) => {
      return matchesRepoFilter(item, filter.repo) &&
        matchesAttentionFilter(item, filter.attention) &&
        matchesQuery(item, ["title", "repo_slug", "source_url", "triage_rationale", "blocked_reason"], filter.query);
    });
  }

  function filteredInvestigations() {
    const filter = filterState("investigations");
    return (state.investigations.items || []).filter((item) => {
      return matchesRepoFilter(item, filter.repo) &&
        matchesAttentionFilter(item, filter.attention) &&
        matchesQuery(item, ["query", "repo_slug", "overall_short_explanation", "run_id", "pause_reason", "pause_until"], filter.query);
    });
  }

  function filteredWorkRuns() {
    const filter = filterState("work");
    return sortRuns(state.overview ? state.overview.work_runs || [] : []).filter((item) => {
      return matchesRepoFilter(item, filter.repo) &&
        matchesAttentionFilter(item, filter.attention) &&
        matchesKindFilter(item, filter.kind) &&
        matchesQuery(item, ["repo_slug", "run_id", "target_url", "current_phase", "status", "backend", "pause_reason", "pause_until"], filter.query);
    });
  }

  function filteredFeedbackItems() {
    const filter = filterState("feedback");
    const sourceItems = state.feedbackTab === "reviews" ? state.feedback.reviews : state.feedback.replies;
    return (sourceItems || []).filter((item) => {
      return matchesRepoFilter(item, filter.repo) &&
        matchesAttentionFilter(item, filter.attention) &&
        matchesQuery(item, ["subject", "repo_slug", "draft_summary", "draft_body", "target_url", "comment_path", "pause_reason", "pause_until"], filter.query);
    });
  }

  function filteredApprovalItems() {
    const filter = filterState("approvals");
    return (state.approvals.items || []).filter((item) => {
      return matchesRepoFilter(item, filter.repo) &&
        matchesAttentionFilter(item, filter.attention) &&
        matchesKindFilter(item, filter.kind) &&
        matchesQuery(item, ["subject", "repo_slug", "reason", "next_action", "target_url"], filter.query);
    });
  }

  function renderApprovalDropButton(item, className = "tx-btn tx-btn-secondary") {
    if (!item) return "";
    let reference = "";
    switch (item.kind) {
      case "work_run":
        reference = item.run_id;
        break;
      case "work_item":
        reference = item.item_id;
        break;
      case "planned_item":
        reference = item.planned_item_id;
        break;
      case "scout_job":
        reference = item.scout_job_id;
        break;
      default:
        return "";
    }
    if (!reference) return "";
    return `<button class="${escapeHTML(className)}" type="button" data-approval-drop-kind="${escapeHTML(item.kind)}" data-approval-drop-id="${escapeHTML(reference)}" data-approval-drop-repo="${escapeHTML(defaultString(item.repo_slug, ""))}" data-approval-drop-subject="${escapeHTML(defaultString(item.subject, ""))}">Drop</button>`;
  }

    if (!item) return '<span class="data-value-text">No action</span>';

        <button class="tx-btn tx-btn-primary" type="button" data-approval-launch-planned="${escapeHTML(item.planned_item_id)}">Approve Launch</button>
        ${renderApprovalDropButton(item)}
      `;
    }
    if (item.action_kind === "retry_scout_job" && item.scout_job_id) {
      return `
        <button class="tx-btn tx-btn-primary" type="button" data-approval-retry-scout="${escapeHTML(item.scout_job_id)}">Retry Scout Job</button>
        ${renderApprovalDropButton(item)}
      `;
    }
    if (item.action_kind === "review_on_github" && item.external_url) {
      return `
        <button class="tx-btn tx-btn-primary" type="button" data-open-external="${escapeHTML(item.external_url)}">Review on GitHub</button>

  function dropApprovalItem(kind, reference, repoSlug, subject) {
    const approvalLabel = defaultString(subject, reference);
    switch (kind) {
      case "work_run":
        requestConfirmation({
          title: "Drop approval item?",
          message: `Remove run ${reference} from the active workspace queue.`,
          confirmLabel: "Drop",
          tone: "danger",
          onConfirm() {
            api(`/api/v1/work/runs/${encodeURIComponent(reference)}/drop`, {
              method: "POST",
            }).then(() => {
              if (state.logSelection.runID === reference && state.logDrawer) {
                state.logDrawer.close();
              }
              load();
              loadApprovals({ silent: true });
              toast("success", "Approval dropped", approvalLabel);
            }).catch((error) => {
              toast("danger", "Drop failed", error.message);
            });
          },
        });
        return;
      case "work_item":
        requestConfirmation({
          title: "Drop approval item?",
          message: `Hide work item ${reference} from the active queue.`,
          confirmLabel: "Drop",
          tone: "danger",
          onConfirm() {
            api(`/api/v1/work-items/${encodeURIComponent(reference)}/drop`, {
              method: "POST",
            }).then(() => {
              load();
              loadApprovals({ silent: true });
              toast("success", "Approval dropped", approvalLabel);
            }).catch((error) => {
              toast("danger", "Drop failed", error.message);
            });
          },
        });
        return;
      case "planned_item":
        requestConfirmation({
          title: "Drop approval item?",
          message: `Remove planned item ${approvalLabel} from the approval queue.`,
          confirmLabel: "Drop",
          tone: "danger",
          onConfirm() {
            api(`/api/v1/planned-items/${encodeURIComponent(reference)}`, {
              method: "DELETE",
            }).then((payload) => {
              if (payload.state) {
                mergeRepoState(payload.state);
              }
              load();
              loadApprovals({ silent: true });
              toast("success", "Approval dropped", defaultString(payload.removed_item && payload.removed_item.title, approvalLabel));
            }).catch((error) => {
              toast("danger", "Drop failed", error.message);
            });
          },
        });
        return;
      case "scout_job":
        if (!repoSlug) {
          toast("warning", "Drop unavailable", "Scout job is missing repo context.");
          return;
        }
        requestConfirmation({
          title: "Drop approval item?",
          message: `Dismiss scout job ${approvalLabel} from the approval queue.`,
          confirmLabel: "Drop",
          tone: "danger",
          onConfirm() {
            mutateScoutItem(repoSlug, reference, "dismiss").then(() => {
              load();
              loadApprovals({ silent: true });
              toast("success", "Approval dropped", approvalLabel);
            }).catch(() => {});
          },
        });
        return;
      default:
        toast("warning", "Drop unavailable", `Approval kind ${defaultString(kind, "(unknown)")} cannot be dropped.`);
    }
  }

  function issueRows(repo) {
    if (!repo || !repo.state) return [];
    return Object.values(repo.state.issues || {}).sort((left, right) => {
      const leftPriority = left.manual_priority_updated_at ? left.manual_priority : (left.priority ?? 5);
      const rightPriority = right.manual_priority_updated_at ? right.manual_priority : (right.priority ?? 5);
      if (leftPriority !== rightPriority) return leftPriority - rightPriority;
      return left.source_number - right.source_number;
    }).map((issue) => ({
      source_number: issue.source_number,
      title: issue.title,
      work_type: workTypeLabel(issue.work_type),
      priority: issue.manual_priority_updated_at ? `P${issue.manual_priority}` : `P${issue.priority ?? 5}`,
      status: defaultString(issue.status, "-"),
      triage: defaultString(issue.triage_status, "-"),
      schedule_at: issue.schedule_at ? formatDateTime(issue.schedule_at) : "Not scheduled",
      deferred_reason: defaultString(issue.deferred_reason, "-"),
    }));
  }

  function serviceTaskRows(repo) {
    if (!repo || !repo.state) return [];
    return Object.values(repo.state.service_tasks || {}).map((task) => ({
      kind: defaultString(task.kind, "-"),

      issue_key: defaultString(task.issue_key || task.planned_item_id, "-"),
      attempts: task.attempts || 0,
      updated_at: task.updated_at ? formatDateTime(task.updated_at) : "No update yet",
      result: defaultString(task.result_summary || task.last_error, "-"),
    }));
  }

  function plannedItemRows(repo) {
    if (!repo || !repo.state) return [];
    return Object.values(repo.state.planned_items || {}).sort((left, right) => {
      return timestamp(right.updated_at) - timestamp(left.updated_at);
    }).map((item) => ({
      id: item.id,
      title: item.title,
      work_type: workTypeLabel(item.work_type),
      launch_kind: defaultString(item.launch_kind, "-"),
      priority: `P${item.priority}`,
      schedule_at: item.schedule_at ? formatDateTime(item.schedule_at) : "Not scheduled",
      state: defaultString(item.state, "-"),
      launch_result: defaultString(item.launch_result || item.last_error, "-"),
    }));
  }

      return;
    }
    const columns = SCHEDULER_BUCKETS.map((bucket) => ({
      id: bucket.id,
      title: bucket.title,

        labels: schedulerCardLabels(item),

      })),
    }));
    replaceWidget(widgetKey, Teryx.kanban("#repo-scheduler-board", {
      columns,
      draggable: true,

        api(`/api/v1/planned-items/${encodeURIComponent(cardId)}`, {
          method: "PATCH",

            schedule_at: schedulerScheduleAtForBucket(toCol),
            clear_schedule: toCol === "ready-now",

        }).catch((error) => {

          renderRepoControlsPage(selectedRepoSummary());
        });
      },
      onCardClick(card) {
        state.repoScheduler.selectedPlannedItemID = card.id;
        renderRepoControlsPage(selectedRepoSummary());

  function populateIssueSelector(issues) {
    const select = document.getElementById("issue-select");
    if (!select) return;
    if (!issues.length) {
      state.selectedIssue = "";
      select.innerHTML = '<option value="">No tracked issues</option>';
      return;
    }
    if (!state.selectedIssue || !issues.some((issue) => String(issue.source_number) === state.selectedIssue)) {
      state.selectedIssue = String(issues[0].source_number);
    }
    select.innerHTML = issues.map((issue) => {
      const selected = String(issue.source_number) === state.selectedIssue ? "selected" : "";
      return `<option value="${issue.source_number}" ${selected}>#${issue.source_number} ${escapeHTML(issue.title)}</option>`;
    }).join("");
  }

  function populatePlannedSelector(items) {
    const select = document.getElementById("planned-select");
    if (!select) return;
    const launchable = items.filter((item) => item.state === "queued" || item.state === "failed");
    if (!launchable.length) {
      select.innerHTML = '<option value="">No launchable items</option>';
      return;
    }
    select.innerHTML = launchable.map((item) => {
      return `<option value="${escapeHTML(item.id)}">${escapeHTML(item.title)} · ${escapeHTML(item.priority)}</option>`;
    }).join("");
  }

  function renderIssueForm(issue) {
    const host = document.getElementById("issue-form");
    if (!host) return;
    destroyForm("issueForm");
    host.innerHTML = "";
    if (!issue) {
      host.innerHTML = '<div class="status-pill">Select a tracked issue to edit.</div>';
      return;
    }
    state.issueForm = Teryx.form("#issue-form", {
      action: "#",
      method: "post",
      layout: "vertical",
      submitLabel: "Save Issue",
      fields: [
        {
          name: "priority",
          label: "Priority",
          type: "select",
          required: true,
          value: String(issue.manual_priority_updated_at ? issue.manual_priority : (issue.priority ?? 5)),
          options: [0, 1, 2, 3, 4, 5].map((priority) => ({ label: `P${priority}`, value: String(priority) })),
        },
        {
          name: "schedule_at",
          label: "Schedule At",
          type: "datetime-local",
          value: toDateTimeLocal(issue.schedule_at),
        },
        {
          name: "deferred_reason",
          label: "Deferred Reason",
          type: "textarea",
          rows: 3,
          value: issue.deferred_reason || "",
        },
      ],
      onSubmit(data) {
        api(`/api/v1/repos/${state.selectedRepo}/issues/${state.selectedIssue}`, {
          method: "PATCH",
          body: JSON.stringify({
            priority: Number(data.priority),
            schedule_at: fromDateTimeLocal(data.schedule_at),
            deferred_reason: data.deferred_reason,
            clear_schedule: !data.schedule_at,
          }),
        }).then((payload) => {
          mergeRepoState(payload.state);
          toast("success", "Issue updated", `Issue #${state.selectedIssue} saved.`);
        }).catch((error) => {
          toast("danger", "Issue update failed", error.message);
        });
        return false;
      },
    });
  }

  function scoutStatusLabel(status) {
    switch (String(status || "").trim()) {
      case "external":
        return "External";
      case "planned":
        return "Planned";
      case "dismissed":
        return "Dismissed";
      default:
        return defaultString(status, "pending").replace(/_/g, " ");
    }
  }

  function scoutActionLabel(action) {
    switch (action) {
      case "retry":
        return "Retry";
      case "dismiss":
        return "Dismiss";
      case "open-run":
        return "Open Run";
      case "reset":
        return "Reset";
      default:
        return action;
    }
  }

  function scoutRoleLabel(roleKey) {
    return repoScoutMeta(String(roleKey || "").replace("-scout", "")).label;
  }

  function workTypeLabel(value) {
    const normalized = String(value || "").trim();
    if (!normalized) return "(untyped)";
    return normalized.replace(/_/g, " ");
  }

  function configStatusPill(enabled) {

  function configFieldName(field) {
    const parts = String(field || "").split(".");
    return parts[parts.length - 1] || "";
  }

  function configFieldOptions(field) {
    switch (configFieldName(field)) {
      case "repo_mode":

          { value: "local", label: "Local" },

          { value: "repo", label: "Repo" },

      case "issue_pick_mode":
        return [
          { value: "manual", label: "Manual" },
          { value: "label", label: "Label" },
          { value: "auto", label: "Auto" },
        ];
      case "pr_forward_mode":
        return [
          { value: "approve", label: "Approve" },
          { value: "auto", label: "Auto" },
        ];
      case "fork_issues_mode":
      case "implement_mode":
        return [
          { value: "manual", label: "Manual" },
          { value: "labeled", label: "Labeled" },
          { value: "auto", label: "Auto" },
        ];
      case "publish_target":
        return [
          { value: "local-branch", label: "Local Branch" },
          { value: "fork", label: "Fork" },
          { value: "repo", label: "Repo" },
        ];
      case "enabled":
        return [
          { value: "true", label: "Enabled" },
          { value: "false", label: "Disabled" },
        ];
      case "mode":
        return [
          { value: "auto", label: "Auto" },
          { value: "manual", label: "Manual" },
        ];

          { value: "local", label: "Local" },
          { value: "repo", label: "Repo" },
          { value: "fork", label: "Fork" },

    }
  }

  function renderConfigSelect(field, label, value, options, disabled = false) {
    const choices = options.map((option) => {
      const selected = String(option.value) === String(value) ? "selected" : "";
      return `<option value="${escapeHTML(option.value)}" ${selected}>${escapeHTML(option.label)}</option>`;
    }).join("");
    return `
      <label class="selector config-selector">
        <span>${escapeHTML(label)}</span>
        <select data-config-field="${escapeHTML(field)}" ${disabled ? "disabled" : ""}>${choices}</select>
      </label>
    `;
  }

  function renderConfigInput(field, label, value, options = {}) {
    const type = options.type || "text";
    const min = options.min !== undefined ? ` min="${escapeHTML(String(options.min))}"` : "";
    const max = options.max !== undefined ? ` max="${escapeHTML(String(options.max))}"` : "";
    const placeholder = options.placeholder ? ` placeholder="${escapeHTML(options.placeholder)}"` : "";
    const disabled = options.disabled ? "disabled" : "";
    return `
      <label class="selector config-selector">
        <span>${escapeHTML(label)}</span>
        <input type="${escapeHTML(type)}" value="${escapeHTML(value)}" data-config-field="${escapeHTML(field)}" ${min}${max}${placeholder} ${disabled} />
      </label>
    `;
  }

  function parseConfigFieldValue(field, value) {
    switch (configFieldName(field)) {
      case "enabled":
        return String(value || "") === "true";
      case "labels":
        return String(value || "")
          .split(",")
          .map((label) => label.trim().toLowerCase())
          .filter(Boolean);
      default:
        return value;
    }
  }

  function configScoutPolicyPath(repo, roleKey) {
    const scouts = repoScoutConfigsByRole(repo);
    const config = scouts && scouts[roleKey];
    return defaultString(config && config.policy_path, "");
  }

  function syncCompatibilityScouts(target) {
    if (!target) return;
    const byRole = target.scouts_by_role || {};
    target.scouts = {
      improvement: byRole.improvement || repoScoutSnapshot("improvement", {}),
      enhancement: byRole.enhancement || repoScoutSnapshot("enhancement", {}),
      ui: byRole.ui || repoScoutSnapshot("ui", {}),
    };
  }

  function setNestedConfigValue(target, field, value) {
    const parts = String(field || "").split(".");
    let cursor = target;
    for (let index = 0; index < parts.length - 1; index += 1) {
      const key = parts[index];
      cursor[key] = { ...(cursor[key] || {}) };
      cursor = cursor[key];
    }
    cursor[parts[parts.length - 1]] = value;
  }

  function renderScoutConfigSection(repo, roleKey, scout) {
    const roleLabel = scoutRoleLabel(roleKey);
    const policyPath = scout.policy_path || configScoutPolicyPath(repo, roleKey);
    const forkDisabled = !scout.enabled || scout.issue_destination !== "fork";
    const sessionLimitField = repoScoutMeta(roleKey, repo).defaultSessionLimit > 0
      ? renderConfigInput(`scouts_by_role.${roleKey}.session_limit`, "Session Limit", String(scout.session_limit || repoScoutMeta(roleKey, repo).defaultSessionLimit), { type: "number", min: 1, max: 6 })
      : "";
    return `
      <div class="config-section">
        <h4>${escapeHTML(roleLabel)}</h4>
        <div class="config-meta">
          <span><strong>Status:</strong> ${escapeHTML(scout.enabled ? "Enabled" : "Disabled")}</span>
          <span><strong>Policy Path:</strong> ${escapeHTML(defaultString(policyPath, "Unavailable"))}</span>
        </div>
        <div class="config-grid">
          ${renderConfigSelect(`scouts_by_role.${roleKey}.enabled`, "Enabled", String(Boolean(scout.enabled)), configFieldOptions("enabled"))}
          ${renderConfigSelect(`scouts_by_role.${roleKey}.mode`, "Mode", scout.mode, configFieldOptions("mode"))}
          ${renderConfigSelect(`scouts_by_role.${roleKey}.schedule`, "Schedule", scout.schedule || defaultString(repoScoutMeta(roleKey, repo).defaultSchedule, "when_resolved"), configFieldOptions("schedule"))}
          ${renderConfigSelect(`scouts_by_role.${roleKey}.issue_destination`, "Destination", scout.issue_destination, configFieldOptions("issue_destination"))}
          ${renderConfigInput(`scouts_by_role.${roleKey}.fork_repo`, "Fork Repo", scout.fork_repo || "", { placeholder: "owner/repo", disabled: forkDisabled })}
          ${sessionLimitField}
          ${renderConfigInput(`scouts_by_role.${roleKey}.labels`, "Labels", (scout.labels || []).join(", "), { placeholder: "comma,separated,labels" })}
        </div>
      </div>
    `;
  }

  function updateConfigDraft(field, rawValue) {
    if (!state.configEditor.draft) return;
    const value = parseConfigFieldValue(field, rawValue);
    const next = cloneConfigSnapshot(state.configEditor.draft);
    setNestedConfigValue(next, field, value);
    switch (field) {
      case "repo_mode":
        state.configEditor.overridePublish = false;
        if (value === "disabled") {
          next.publish_target = "";
        } else {
          next.publish_target = repoModeToPublishTarget(value);
        }
        break;
      case "publish_target":
        state.configEditor.overridePublish = true;
        next.repo_mode = publishTargetToRepoMode(value);
        break;
      case "issue_pick_mode": {
        const automationMode = issuePickModeToAutomationMode(value);
        state.configEditor.overrideForkIssues = false;
        state.configEditor.overrideImplement = false;
        next.fork_issues_mode = automationMode;
        next.implement_mode = automationMode;
        break;
      }
      case "fork_issues_mode":
        state.configEditor.overrideForkIssues = true;
        next.issue_pick_mode = effectiveIssuePickFromAdvanced(next);
        break;
      case "implement_mode":
        state.configEditor.overrideImplement = true;
        next.issue_pick_mode = effectiveIssuePickFromAdvanced(next);
        break;
      default:
        if (field.startsWith("scouts_by_role.")) {
          const parts = field.split(".");
          const roleKey = parts[1];
          const scout = (next.scouts_by_role && next.scouts_by_role[roleKey]) || null;
          if (scout) {
            scout.policy_path = configScoutPolicyPath(selectedRepoSummary(), roleKey);
            if (parts[2] === "issue_destination" && value !== "fork") {
              scout.fork_repo = "";
            }
          }
        }
        break;
    }
    syncCompatibilityScouts(next);
    state.configEditor.draft = next;
    state.configEditor.dirty = !configDraftEquals(next, state.configEditor.server);
    if (!state.configEditor.dirty) {
      state.configEditor.remoteChanged = false;
      state.configEditor.overridePublish = false;
      state.configEditor.overrideForkIssues = false;
      state.configEditor.overrideImplement = false;
    }
    renderRepoConfigPage(selectedRepoSummary());
  }

  function resetConfigDraft(repo) {
    resetConfigEditorForRepo(repo);
    renderRepoConfigPage(repo);
  }

  function saveConfigDraft(repo) {
    if (!repo || !state.configEditor.draft || state.configEditor.saving) {
      return;
    }
    state.configEditor.saving = true;
    renderRepoConfigPage(repo);
    api(`/api/v1/repos/${repo.repo_slug}/settings`, {
      method: "PATCH",
      body: JSON.stringify(state.configEditor.draft),
    }).then((payload) => {
      state.configEditor.saving = false;
      mergeRepoSummary(payload.repo);
      toast("success", "Repo config updated", repo.repo_slug);
    }).catch((error) => {
      state.configEditor.saving = false;
      renderRepoConfigPage(repo);
      toast("danger", "Repo config failed", error.message);
    });
  }

  function refreshCounts(repo, updatedState) {
    repo.issue_counts = {};
    repo.service_task_counts = {};
    repo.scout_job_counts = {};
    repo.planned_item_counts = {};
    Object.values(updatedState.issues || {}).forEach((issue) => {
      const key = issue.status || "";
      repo.issue_counts[key] = (repo.issue_counts[key] || 0) + 1;
    });
    Object.values(updatedState.service_tasks || {}).forEach((task) => {
      const key = task.status || "";
      repo.service_task_counts[key] = (repo.service_task_counts[key] || 0) + 1;
    });
    Object.values(updatedState.scout_jobs || {}).forEach((job) => {
      const key = job.status || "";
      repo.scout_job_counts[key] = (repo.scout_job_counts[key] || 0) + 1;
    });
    Object.values(updatedState.planned_items || {}).forEach((item) => {
      const key = item.state || "";
      repo.planned_item_counts[key] = (repo.planned_item_counts[key] || 0) + 1;
    });
  }

  function mergeRepoSummary(updatedRepo, options = {}) {
    if (!state.overview || !updatedRepo) return;
    const index = state.overview.repos.findIndex((entry) => entry.repo_slug === updatedRepo.repo_slug);
    if (index === -1) return;
    const existing = state.overview.repos[index];
    state.overview.repos[index] = {
      ...existing,
      ...updatedRepo,
      state: updatedRepo.state || existing.state || null,
    };
    if (state.configEditor.repoSlug === updatedRepo.repo_slug) {
      resetConfigEditorForRepo(state.overview.repos[index]);
    }
    if (options.resetScouts && state.scoutItems.repoSlug === updatedRepo.repo_slug) {
      resetScoutItemsState(state.overview.repos[index]);
    }
    if (options.rerender !== false) {
      renderApp();
    }
  }

  function mergeRepoState(updatedState) {
    if (!state.overview) return;
    const repo = state.overview.repos.find((entry) => entry.repo_slug === updatedState.source_repo);
    if (!repo) return;
    repo.state = updatedState;
    repo.updated_at = updatedState.updated_at;
    refreshCounts(repo, updatedState);
    reconcileRepoScheduler(repo);
    renderApp();
  }

  function openLogDrawer(runID) {
    initLogDrawer();
    state.logSelection = defaultLogSelection();
    state.logSelection.runID = runID;
    renderLogDrawer();
    state.logDrawer.open();
    loadRunDetail(runID);
    loadLogMetadata(runID);

  function loadRunDetail(runID) {
    state.logSelection.runID = runID;
    state.logSelection.detailLoading = true;
    state.logSelection.error = "";
    renderLogDrawer();
    return api(`/api/v1/work/runs/${encodeURIComponent(runID)}`).then((payload) => {
      if (state.logSelection.runID !== runID) return;
      state.logSelection.detail = payload;
      state.logSelection.detailLoading = false;
      renderLogDrawer();
    }).catch((error) => {
      if (state.logSelection.runID !== runID) return;
      state.logSelection.detailLoading = false;
      state.logSelection.error = error.message;
      renderLogDrawer();
      toast("danger", "Run detail failed", error.message);
    });
  }

  function loadLogMetadata(runID, options = {}) {
    state.logSelection.runID = runID;
    state.logSelection.loading = true;
    state.logSelection.error = "";
    renderLogDrawer();
    return api(`/api/v1/work/runs/${encodeURIComponent(runID)}/logs`).then((payload) => {
      if (state.logSelection.runID !== runID) return;
      state.logSelection.meta = payload;
      state.logSelection.loading = false;
      const requestedPath = options.preserveFile ? state.logSelection.filePath : "";
      const available = (payload.files || []).map((file) => file.path);
      const filePath = available.includes(requestedPath)
        ? requestedPath
        : (payload.default_path || available[0] || "");
      state.logSelection.filePath = filePath;
      renderLogDrawer();
      if (filePath) {
        return loadLogContent(runID, filePath);
      }
    }).catch((error) => {
      if (state.logSelection.runID !== runID) return;
      state.logSelection.loading = false;
      state.logSelection.error = error.message;
      renderLogDrawer();
      toast("danger", "Log load failed", error.message);
    });
  }

  function loadLogContent(runID, filePath) {
    state.logSelection.filePath = filePath;
    state.logSelection.loadingContent = true;
    renderLogDrawer();
    const query = new URLSearchParams({ path: filePath, tail: "200" });
    return api(`/api/v1/work/runs/${encodeURIComponent(runID)}/logs/content?${query.toString()}`).then((payload) => {
      if (state.logSelection.runID !== runID || state.logSelection.filePath !== filePath) return;
      state.logSelection.loadingContent = false;
      state.logSelection.content = payload.content || "";
      state.logSelection.error = "";
      if (state.logSelection.meta && payload.summary) {
        state.logSelection.meta.summary = payload.summary;
      }
      renderLogDrawer();
    }).catch((error) => {
      if (state.logSelection.runID !== runID) return;
      state.logSelection.loadingContent = false;
      state.logSelection.error = error.message;
      renderLogDrawer();
      toast("danger", "Log content failed", error.message);
    });
  }

  function refreshOpenLogDrawerIfNeeded() {
    if (!state.logSelection.runID || !state.logSelection.meta) return;
    const current = currentRunSummary(state.logSelection.runID);
    if (!current) return;
    const previous = state.logSelection.meta.summary || {};
    if (current.updated_at !== previous.updated_at || current.status !== previous.status || current.current_phase !== previous.current_phase) {
      loadRunDetail(state.logSelection.runID);
      loadLogMetadata(state.logSelection.runID, { preserveFile: true });
    }
  }

  function groupInlineCommentsByPath(comments) {
    const grouped = {};
    (comments || []).forEach((comment) => {
      const key = defaultString(comment.path, "(no path)");
      if (!grouped[key]) {
        grouped[key] = [];
      }
      grouped[key].push(comment);
    });
    return grouped;
  }

  function renderReviewDraftDetail(draft) {
    if (!draft) {
      return '<div class="status-pill">No draft is ready yet.</div>';
    }
    const grouped = groupInlineCommentsByPath(draft.inline_comments || []);
    const groupedHTML = Object.keys(grouped).length
      ? Object.entries(grouped).map(([path, comments]) => `
          <article class="queue-item">
            <span class="queue-label">${escapeHTML(path)}</span>
            <ul class="status-list">
              ${comments.map((comment) => `
                <li>
                  <strong>Line ${escapeHTML(defaultString(comment.line, "-"))}</strong>
                  <span>${escapeHTML(defaultString(comment.body, "(none)"))}</span>
                </li>
              `).join("")}
            </ul>
          </article>
        `).join("")
      : '<div class="status-pill">No inline comments were generated.</div>';
    return `
      <div class="queue-summary">
        <article class="queue-item"><span class="queue-label">Summary</span><p class="queue-copy">${escapeHTML(defaultString(draft.summary, "(none)"))}</p></article>
        <article class="queue-item"><span class="queue-label">Review Event</span><p class="queue-copy">${escapeHTML(defaultString(draft.review_event, "(none)"))}</p></article>
        <article class="queue-item"><span class="queue-label">Disposition</span><p class="queue-copy">${escapeHTML(defaultString(draft.suggested_disposition, "(none)"))}</p></article>

  function attachHandlers() {
    document.getElementById("refresh-button").addEventListener("click", () => {
      load();
      if (state.currentView !== "usage") {
        refreshCurrentView({ silent: true });
      }
      if (state.currentView === "repo" && state.repoTab === "scouts" && state.selectedRepo) {
        loadScoutItems(state.selectedRepo, { silent: true });
      }
      if (state.logSelection.runID) {
        loadRunDetail(state.logSelection.runID);
        loadLogMetadata(state.logSelection.runID, { preserveFile: true });
      }
    });

    document.getElementById("quick-switch-button").addEventListener("click", () => {
      runQuickSwitch();
    });

    document.getElementById("quick-switch-input").addEventListener("keydown", (event) => {
      if (event.key === "Enter") {
        event.preventDefault();
        runQuickSwitch();
      }
    });

    document.getElementById("sidebar-nav").addEventListener("click", (event) => {
      if (event.target.closest("[data-view-filter-reset]")) {
        event.preventDefault();
        event.stopPropagation();
        return;
      }
      const button = event.target.closest("[data-nav-view]");
      if (!button) return;
      const nextView = button.getAttribute("data-nav-view");
      if (nextView && nextView !== "repo") {
        state.currentView = nextView;
        ensureViewData(nextView);
        renderApp();
        return;
      }
      const repoSlug = button.getAttribute("data-repo-slug");
      if (!repoSlug) return;
      openRepoView(repoSlug, "overview");
    });

    document.getElementById("sidebar-nav").addEventListener("input", (event) => {
      if (!event.target.matches("[data-sidebar-query]")) return;
      const caret = typeof event.target.selectionStart === "number" ? event.target.selectionStart : null;
      state.sidebarQuery = String(event.target.value || "");
      renderSidebar();
      const nextInput = document.querySelector("[data-sidebar-query]");
      if (nextInput) {
        nextInput.focus();
        if (caret !== null && typeof nextInput.setSelectionRange === "function") {
          nextInput.setSelectionRange(caret, caret);
        }
      }
    });

    document.getElementById("page-tabs").addEventListener("click", (event) => {
      const button = event.target.closest("[data-tab-group]");
      if (!button) return;
      const group = button.getAttribute("data-tab-group");
      const tabID = button.getAttribute("data-tab-id") || "";
      if (group === "repo") {
        state.repoTab = tabID || "overview";
        renderApp();
        return;
      }
      if (group === "feedback") {

      const tab = event.target.closest("[data-tab-group]");
      if (!tab) return;
      if (!["ArrowLeft", "ArrowRight", "Home", "End"].includes(event.key)) return;
      event.preventDefault();
      const group = tab.getAttribute("data-tab-group");
      const tabs = Array.from(document.querySelectorAll(`[data-tab-group="${group}"]`));
      const currentIndex = tabs.indexOf(tab);
      if (currentIndex === -1) return;
      let nextIndex = currentIndex;
      if (event.key === "ArrowRight") {
        nextIndex = (currentIndex + 1) % tabs.length;
      } else if (event.key === "ArrowLeft") {
        nextIndex = (currentIndex - 1 + tabs.length) % tabs.length;
      } else if (event.key === "Home") {
        nextIndex = 0;
      } else if (event.key === "End") {
        nextIndex = tabs.length - 1;
      }
      tabs[nextIndex].focus();
      tabs[nextIndex].click();
    });

    document.getElementById("page-body").addEventListener("submit", (event) => {
      const form = event.target.closest("#usage-filters-form");
      if (!form) return;
      event.preventDefault();
      state.usage.filters = {
        since: defaultString(document.getElementById("usage-filter-since") && document.getElementById("usage-filter-since").value, "30d"),
        root: defaultString(document.getElementById("usage-filter-root") && document.getElementById("usage-filter-root").value, "all"),
        activity: String(document.getElementById("usage-filter-activity") && document.getElementById("usage-filter-activity").value || "").trim(),
        phase: String(document.getElementById("usage-filter-phase") && document.getElementById("usage-filter-phase").value || "").trim(),
        model: String(document.getElementById("usage-filter-model") && document.getElementById("usage-filter-model").value || "").trim(),
      };
      loadUsage();
    });

    document.getElementById("page-body").addEventListener("click", (event) => {
      const resetButton = event.target.closest("[data-view-filter-reset]");
      if (!resetButton) return;
      event.preventDefault();
      event.stopPropagation();
      const view = resetButton.getAttribute("data-view-filter-reset") || "";
      if (!view) return;
      resetFilter(view);
      if (view === "issues") {
        renderIssuesPage();
      } else if (view === "investigations") {
        renderInvestigationsPage();
      } else if (view === "work") {
        renderWorkPage();
      } else if (view === "feedback") {
        renderFeedbackPage();
      } else if (view === "approvals") {
        renderApprovalsPage();
      }
    });

    document.getElementById("page-body").addEventListener("click", (event) => {
      const resetButton = event.target.closest("[data-usage-reset]");
      if (!resetButton) return;
      event.preventDefault();
      state.usage = defaultUsageState();
      loadUsage();
    });

    document.getElementById("confirm-dialog-cancel").addEventListener("click", () => {
      closeConfirmationDialog();
    });

    document.getElementById("confirm-dialog-confirm").addEventListener("click", (event) => {
      event.preventDefault();
      const handler = state.confirmDialog.onConfirm;
      closeConfirmationDialog();
      if (typeof handler === "function") {
        handler();
      }
    });

    document.getElementById("confirm-dialog").addEventListener("cancel", (event) => {
      event.preventDefault();
      closeConfirmationDialog();
    });

    window.addEventListener("hashchange", () => {
      applyHashState();
      if (state.currentView !== "repo") {
        ensureViewData(state.currentView);
      }
      renderApp();
    });

    document.addEventListener("visibilitychange", () => {
      if (document.visibilityState === "visible") {
        refreshUsageIfStale();
        return;
      }
      syncUsageAutoRefresh();
    });

    document.body.addEventListener("change", (event) => {
      if (event.target.id === "log-file-select") {
        const filePath = String(event.target.value || "");
        if (state.logSelection.runID && filePath) {
          loadLogContent(state.logSelection.runID, filePath);
        }
      }
    });

    document.body.addEventListener("click", (event) => {
      const button = event.target.closest("[data-log-refresh]");
      if (!button || !state.logSelection.runID) return;
      loadRunDetail(state.logSelection.runID);
      loadLogMetadata(state.logSelection.runID, { preserveFile: true });
    });

    document.body.addEventListener("click", (event) => {
      const runButton = event.target.closest("[data-work-item-run]");
      if (runButton) {
        const itemID = runButton.getAttribute("data-work-item-run");
        api(`/api/v1/work-items/${encodeURIComponent(itemID)}/run`, {
          method: "POST",
        }).then(() => {
          toast("success", "Work item started", itemID);
          load();
          loadWorkItemDetail(itemID);
        }).catch((error) => {
          toast("danger", "Run failed", error.message);
        });
        return;
      }

      const submitButton = event.target.closest("[data-work-item-submit]");
      if (submitButton) {
        const itemID = submitButton.getAttribute("data-work-item-submit");
        requestConfirmation({
          title: "Submit draft?",
          message: `Submit work item ${itemID} to its external target.`,
          confirmLabel: "Submit Draft",
          onConfirm() {
            api(`/api/v1/work-items/${encodeURIComponent(itemID)}/submit`, {
              method: "POST",
            }).then(() => {
              toast("success", "Work item submitted", itemID);
              load();
              loadWorkItemDetail(itemID);
            }).catch((error) => {
              toast("danger", "Submit failed", error.message);
            });
          },
        });
        return;
      }

      const dropButton = event.target.closest("[data-work-item-drop]");
      if (dropButton) {
        const itemID = dropButton.getAttribute("data-work-item-drop");
        requestConfirmation({
          title: "Drop work item?",
          message: `Hide work item ${itemID} from the active queue.`,
          confirmLabel: "Drop Item",
          tone: "danger",
          onConfirm() {
            api(`/api/v1/work-items/${encodeURIComponent(itemID)}/drop`, {
              method: "POST",
            }).then(() => {
              toast("success", "Work item dropped", itemID);
              load();
              loadWorkItemDetail(itemID);
            }).catch((error) => {
              toast("danger", "Drop failed", error.message);
            });
          },
        });
        return;
      }

      const restoreButton = event.target.closest("[data-work-item-restore]");
      if (restoreButton) {
        const itemID = restoreButton.getAttribute("data-work-item-restore");
        api(`/api/v1/work-items/${encodeURIComponent(itemID)}/restore`, {
          method: "POST",
        }).then(() => {
          toast("success", "Work item restored", itemID);
          load();
          loadWorkItemDetail(itemID);
        }).catch((error) => {
          toast("danger", "Restore failed", error.message);
        });
        return;
      }

      const fixButton = event.target.closest("[data-work-item-fix]");
      if (fixButton) {
        const itemID = fixButton.getAttribute("data-work-item-fix");
        const instructionInput = document.getElementById("work-item-fix-instruction");
        const instruction = instructionInput
          ? String(instructionInput.value || "").trim()
          : String(state.workItemSelection.fixInstruction || "").trim();
        if (!instruction) return;
        api(`/api/v1/work-items/${encodeURIComponent(itemID)}/fix`, {
          method: "POST",
          body: JSON.stringify({ instruction }),
        }).then(() => {
          toast("success", "Work item draft updated", itemID);
          load();
          loadWorkItemDetail(itemID);
        }).catch((error) => {
          toast("danger", "Fix failed", error.message);
        });
        return;
      }

      const refreshButton = event.target.closest("[data-work-item-refresh]");
      if (refreshButton) {
        const itemID = refreshButton.getAttribute("data-work-item-refresh");
        loadWorkItemDetail(itemID);
      }
    });
  }

  function connectEvents() {
    const source = new EventSource(apiBase + "/api/v1/events");
    source.addEventListener("state", (event) => {
      const payload = JSON.parse(event.data);
      const previousGeneratedAt = state.liveVersions.generatedAt;
      const previousOverviewVersion = state.liveVersions.overview;
      const previousUsageVersion = state.liveVersions.usage;
      const currentOverviewGeneratedAt = defaultString(state.overview && state.overview.generated_at, "");
      const currentUsageVersion = defaultString(state.usage.report && state.usage.report.version, "");
      state.liveVersions = {
        generatedAt: defaultString(payload.generated_at, previousGeneratedAt || new Date().toISOString()),
        overview: defaultString(payload.overview_version, previousOverviewVersion),
        usage: defaultString(payload.usage_version, previousUsageVersion),
      };
      const overviewChanged = Boolean(
        (previousOverviewVersion && state.liveVersions.overview && previousOverviewVersion !== state.liveVersions.overview)
        || (!previousOverviewVersion && currentOverviewGeneratedAt && currentOverviewGeneratedAt !== state.liveVersions.generatedAt)
      );
      const usageChanged = Boolean(
        (previousUsageVersion && state.liveVersions.usage && previousUsageVersion !== state.liveVersions.usage)
        || (!previousUsageVersion && currentUsageVersion && state.liveVersions.usage && currentUsageVersion !== state.liveVersions.usage)
      );
      if (overviewChanged) {
        loadOverview({ silent: true });
      }
      if (state.currentView === "usage" && usageChanged) {
        loadUsage({ silent: true });
      }
      if (overviewChanged || usageChanged || (previousGeneratedAt && previousGeneratedAt !== state.liveVersions.generatedAt)) {
        announce("Workspace data updated.");
      }
    });
    source.onerror = () => {
      source.close();
      setTimeout(connectEvents, 4000);
    };
  }

  function load() {
    if (state.currentView === "usage") {
      state.usage.loading = true;
      state.usage.error = "";
      renderApp();
      return Promise.allSettled([
        loadOverview(),
        loadRepos({ silent: true }),
        loadUsage({ silent: true }),
      ]);
    }
    return Promise.allSettled([
      loadOverview(),
      loadRepos({ silent: true }),
    ]).then(() => refreshCurrentView({ silent: true })).catch(() => {});
  }

  initLogDrawer();
  initWorkItemDrawer();
  applyHashState();
  state.hashReady = true;
  attachHandlers();
  load().then(connectEvents);
})();
